CREATE PROGRAM CCL_SCREEN1

SELECT	ACT = EVALUATE(P.ACTIVE_IND,1,"ACTIVE",2,"INACTIVE"),
	P.PERSON_ID,
	DEPT = UAR_GET_CODE_DISPLAY(P.DEPARTMENT_CD),
	POSITION = UAR_GET_CODE_DISPLAY(P.POSITION_CD),
	P.NAME_FULL_FORMATTED,
	P.USERNAME,
	P.EMAIL,
	P.PHYSICIAN_IND,
	P.UPDT_DT_TM
FROM	PRSNL P
WHERE	P.USERNAME = "ABC123"

WITH NOCOUNTER*

END
GO

This program selects data from the PRSNL table qualifying on the username.
